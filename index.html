@Column(length = 20) private String status = "EN_ATTENTE"; // Statut initial du ticket private String qrCodeToken; // Jeton unique généré pour le scan // --- CONSTRUCTEURS --- public Appointment() {} public Appointment(String clientName, String email, String reason, LocalDateTime appointmentTime) { this.clientName = clientName; } public void setId(Long id) { this.id = id; } public String getClientName() { return clientName; } public void setClientName(String clientName) { this.clientName = clientName; } public String getEmail() { return email; } public void setEmail(String email) { this.email = email; } public String getReason() { return reason; } public void setReason(String reason) { this.reason = reason; } public LocalDateTime getAppointmentTime() { return appointmentTime; } public void setAppointmentTime(LocalDateTime appointmentTime) { this.appointmentTime = appointmentTime; } public String getStatus() { return status; } public void setStatus(String status) { this.status = status; } public String getQrCodeToken() { return qrCodeToken; } } // =============================================================== // FICHIER 2 : Le Contrôleur (La logique de pilotage) // Chemin : src/main/java/com/fnac/sav/controller/AppointmentController.java // =============================================================== package com.fnac.sav.controller;
